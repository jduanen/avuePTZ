# install packages that can't be pip-installed on Trixie
* `sudo apt install python3-systemd python3-serial`

# create a venv that can see these packages
* `python3 -m venv --system-site-packages venv`
* `source venv/bin/activate`

# install remaining packages
* `pip install -r requirements.txt`

# subsequently, activated venv
* `cd ~/Code/avuePTZ`
* `source venv/bin/activate`

# install and enable avuePTZ service
* `cd ${HOME}/Code/avuePTZ/`
* `sudo cp etc/systemd/system/avuePTZ.service /etc/systemd/system/`
* `sudo systemctl daemon-reload`
* `sudo systemctl enable avuePTZ`
* `sudo systemctl start avuePTZ`
* `sudo systemctl status avuePTZ`  # check that it's running correctly
